#Crie um programa que recebe um número e calcula uma fatorial (acrescentei repetiçoes caso não estivesse dentro do valor esperado.)

numero = int(input('Digite um número:'))

acerto = False
while acerto == False:
  if numero < 0:
    print ('Número inválido')
    numero = int(input('Digite um número:'))
  else:
   acerto = True
   fatorial = 1
   for num in range(1,numero +1):
    fatorial = fatorial * num
    print(fatorial)
